document.addEventListener('DOMContentLoaded', function() { const calendarEl = document.getElementById('calendar'); const modalEl = document.getElementById('appointmentModal'); const modal = new bootstrap.Modal(modalEl); const form = document.getElementById('appointmentForm'); const btnDelete = document.getElementById('btnDelete'); const inputStart = document.getElementById('appointmentStart'); const inputEnd = document.getElementById('appointmentEnd'); const selectServico = document.getElementById('eventColor'); const themeBtn = document.getElementById('themeBtn'); const themeIcon = document.getElementById('themeIcon'); const savedTheme = localStorage.getItem('theme_nath') || 'light'; document.body.setAttribute('data-theme', savedTheme); themeBtn.addEventListener('click', () => { const newTheme = document.body.getAttribute('data-theme') === 'light' ? 'dark' : 'light'; document.body.setAttribute('data-theme', newTheme); localStorage.setItem('theme_nath', newTheme); themeIcon.className = newTheme === 'light' ? 'bi bi-moon-stars fs-4' : 'bi bi-sun fs-4'; }); function calcularFim() { if (inputStart.value) { const selectedOption = selectServico.options[selectServico.selectedIndex]; const duracaoEmMinutos = parseInt(selectedOption.getAttribute('data-duration')) || 60; let start = new Date(inputStart.value); let end = new Date(start.getTime() + (duracaoEmMinutos * 60 * 1000)); inputEnd.value = new Date(end - (end.getTimezoneOffset() * 60000)).toISOString().slice(0, 16); } } inputStart.addEventListener('change', calcularFim); selectServico.addEventListener('change', calcularFim); const calendar = new FullCalendar.Calendar(calendarEl, { initialView: window.innerWidth < 768 ? 'timeGridDay' : 'timeGridWeek', locale: 'pt-br', nowIndicator: true, allDaySlot: false, slotMinTime: '07:00:00', slotMaxTime: '21:00:00', height: '100%', titleFormat: { year: 'numeric', month: '2-digit', day: '2-digit' }, selectable: true, editable: true, unselectAuto: true, longPressDelay: 50, headerToolbar: { left: 'prev,next', center: 'title', right: 'today,dayGridMonth,timeGridWeek,timeGridDay,listWeek' }, buttonText: { today: 'Hoje', month: 'Mês', week: 'Semana', day: 'Dia', list: 'Lista' }, select: (info) => { openNewModal(info.startStr.slice(0, 16)); calendar.unselect(); }, eventClick: (info) => editEvent(info.event), eventDrop: () => updateStorage(), eventResize: () => updateStorage() }); calendar.render(); // FUNÇÃO CORRIGIDA COM ASYNC async function updateStorage() { const events = calendar.getEvents().map(e => ({ id: e.id, title: e.title, start: e.startStr, end: e.endStr, backgroundColor: e.backgroundColor })); const stringData = JSON.stringify(events); localStorage.setItem('appointments_nath', stringData); // Envia para o Render try { await fetch("https://nathricco.onrender.com/salvar-agenda", { method: "POST", headers: { "Content-Type": "application/json" }, body: stringData }); console.log("Sincronizado com sucesso!"); } catch (e) { console.error("Erro ao sincronizar com servidor:", e); } } async function loadEvents() { const localData = localStorage.getItem('appointments_nath'); if (localData) { calendar.removeAllEvents(); calendar.addEventSource(JSON.parse(localData)); } try { // Busca o arquivo atualizado do GitHub (via GitHub Pages ou API) const res = await fetch("https://azdevcoder.github.io/nathricco/dados/agendamentos.json?t=" + Date.now()); if (res.ok) { const serverData = await res.json(); const serverString = JSON.stringify(serverData); if (serverString !== localData) { calendar.removeAllEvents(); calendar.addEventSource(serverData); localStorage.setItem('appointments_nath', serverString); } } } catch (e) { console.log("Usando dados locais."); } } loadEvents(); window.openNewModal = (date = '') => { form.reset(); document.getElementById('eventID').value = ''; inputStart.value = date || new Date(new Date().getTime() - new Date().getTimezoneOffset() * 60000).toISOString().slice(0, 16); calcularFim(); btnDelete.classList.add('d-none'); modal.show(); }; function editEvent(event) { document.getElementById('eventID').value = event.id; document.getElementById('patientName').value = event.title.split(' - ')[0]; inputStart.value = event.startStr.slice(0, 16); inputEnd.value = event.endStr ? event.endStr.slice(0, 16) : ''; document.getElementById('eventColor').value = event.backgroundColor; btnDelete.classList.remove('d-none'); modal.show(); } form.onsubmit = async (e) => { e.preventDefault(); const id = document.getElementById('eventID').value; const servicoNome = selectServico.options[selectServico.selectedIndex].text; const eventData = { id: id || "ID-" + Date.now(), title: document.getElementById('patientName').value + " - " + servicoNome, start: inputStart.value, end: inputEnd.value, backgroundColor: selectServico.value }; if (id) { const ev = calendar.getEventById(id); if (ev) ev.remove(); } calendar.addEvent(eventData); modal.hide(); mostrarNotificacao(); await updateStorage(); }; btnDelete.onclick = async () => { const id = document.getElementById('eventID').value; const patientName = document.getElementById('patientName').value; if (id && confirm(`Excluir agendamento de ${patientName}?`)) { const ev = calendar.getEventById(id); if (ev) { ev.remove(); modal.hide(); mostrarNotificacao(); await updateStorage(); } } }; function mostrarNotificacao() { const toast = document.getElementById('toastSuccess'); toast.style.display = 'block'; setTimeout(() => { toast.style.display = 'none'; }, 3000); } });
